  5 /**
  6  * A Transform in 3-space. Values are copied into this object.
  7  *
  8  *  @constructor
  9  *  @param options - object with following keys:
 10  *   * translation - the Vector3 describing the translation
 11  *   * rotation - the ROSLIB.Quaternion describing the rotation
 12  */
 13 ROSLIB.Transform = function(options) {
 14   var options = options || {};
 15   // copy the values into this object if they exist
 16   this.translation = new ROSLIB.Vector3(options.translation);
 17   this.rotation = new ROSLIB.Quaternion(options.rotation);
 18 };
 19 
 20 /**
 21  * Clone a copy of this transform.
 22  *
 23  * @returns the cloned transform
 24  */
 25 ROSLIB.Transform.prototype.clone = function() {
 26   return new ROSLIB.Transform(this);
 27 };
