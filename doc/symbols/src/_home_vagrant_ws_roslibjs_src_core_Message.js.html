  5 /**
  6  * Message objects are used for publishing and subscribing to and from topics.
  7  *
  8  * @constructor
  9  * @param values - object matching the fields defined in the .msg definition file.
 10  */
 11 ROSLIB.Message = function(values) {
 12   var message = this;
 13   if (values) {
 14     Object.keys(values).forEach(function(name) {
 15       message[name] = values[name];
 16     });
 17   }
 18 };
