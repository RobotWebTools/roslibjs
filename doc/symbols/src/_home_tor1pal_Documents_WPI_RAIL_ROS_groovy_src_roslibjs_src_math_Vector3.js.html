  5 /**
  6  * A 3D vector.
  7  *
  8  *  @constructor
  9  *  @param options - object with following keys:
 10  *   * x - the x value 
 11  *   * y - the y value 
 12  *   * z - the z value 
 13  */
 14 ROSLIB.Vector3 = function(options) {
 15   var options = options || {};
 16   this.x = options.x || 0;
 17   this.y = options.y || 0;
 18   this.z = options.z || 0;
 19 };
 20 
 21 /**
 22  * Set the values of this vector to the sum of itself and the given vector.
 23  *
 24  * @param v the vector to add with
 25  */
 26 ROSLIB.Vector3.prototype.add = function(v) {
 27   this.x += v.x;
 28   this.y += v.y;
 29   this.z += v.z;
 30 };
 31 
 32 /**
 33  * Set the values of this vector to the difference of itself and the given vector.
 34  *
 35  * @param v the vector to subtract with
 36  */
 37 ROSLIB.Vector3.prototype.subtract = function(v) {
 38   this.x -= v.x;
 39   this.y -= v.y;
 40   this.z -= v.z;
 41 };
 42 
 43 /**
 44  * Multiply the given Quaternion with this vector.
 45  *
 46  * @param q - the quaternion to multiply with
 47  */
 48 ROSLIB.Vector3.prototype.multiplyQuaternion = function(q) {
 49   var ix = q.w * this.x + q.y * this.z - q.z * this.y;
 50   var iy = q.w * this.y + q.z * this.x - q.x * this.z;
 51   var iz = q.w * this.z + q.x * this.y - q.y * this.x;
 52   var iw = -q.x * this.x - q.y * this.y - q.z * this.z;
 53   this.x = ix * q.w + iw * -q.x + iy * -q.z - iz * -q.y;
 54   this.y = iy * q.w + iw * -q.y + iz * -q.x - ix * -q.z;
 55   this.z = iz * q.w + iw * -q.z + ix * -q.y - iy * -q.x;
 56 };
 57 
 58 /**
 59  * Clone a copy of this vector.
 60  *
 61  * @returns the cloned vector
 62  */
 63 ROSLIB.Vector3.prototype.clone = function() {
 64   return new ROSLIB.Vector3(this);
 65 };
