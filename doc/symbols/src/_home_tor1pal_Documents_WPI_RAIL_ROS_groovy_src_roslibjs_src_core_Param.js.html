  5 /**
  6  * A ROS parameter.
  7  *
  8  * @constructor
  9  * @param options - possible keys include:
 10  *   * name - the param name, like max_vel_x
 11  */
 12 ROSLIB.Param = function(options) {
 13   var param = this;
 14   options = options || {};
 15   param.name = options.name;
 16 
 17   /**
 18    * Fetches the value of the param.
 19    *
 20    * @param callback - function with the following params:
 21    *  * value - the value of the param from ROS.
 22    */
 23   param.get = function(callback) {
 24     var paramClient = new ROSLIB.Service({
 25       name : '/rosapi/get_param',
 26       serviceType : 'rosapi/GetParam'
 27     });
 28 
 29     var request = new ROSLIB.ServiceRequest({
 30       name : param.name,
 31       value : JSON.stringify('')
 32     });
 33 
 34     paramClient.callService(request, function(result) {
 35       var value = JSON.parse(result.value);
 36       callback(value);
 37     });
 38   };
 39 
 40   /**
 41    * Sets the value of the param in ROS.
 42    *
 43    * @param value - value to set param to.
 44    */
 45   param.set = function(value) {
 46     var paramClient = new ROSLIB.Service({
 47       name : '/rosapi/set_param',
 48       serviceType : 'rosapi/SetParam'
 49     });
 50 
 51     var request = new ROSLIB.ServiceRequest({
 52       name : param.name,
 53       value : JSON.stringify(value)
 54     });
 55 
 56     paramClient.callService(request, function() {
 57     });
 58   };
 59 };
