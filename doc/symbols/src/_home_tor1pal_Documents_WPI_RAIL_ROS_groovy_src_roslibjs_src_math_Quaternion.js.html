  5 /**
  6  * A Quaternion.
  7  *
  8  *  @constructor
  9  *  @param options - object with following keys:
 10  *   * x - the x value 
 11  *   * y - the y value 
 12  *   * z - the z value 
 13  *   * w - the w value 
 14  */
 15 ROSLIB.Quaternion = function(options) {
 16   var options = options || {};
 17   this.x = options.x || 0;
 18   this.y = options.y || 0;
 19   this.z = options.z || 0;
 20   this.w = options.w || 1;
 21 };
 22 
 23 /**
 24  * Perform a conjugation on this quaternion.
 25  */
 26 ROSLIB.Quaternion.prototype.conjugate = function() {
 27   this.x *= -1;
 28   this.y *= -1;
 29   this.z *= -1;
 30 };
 31 
 32 /**
 33  * Perform a normalization on this quaternion.
 34  */
 35 ROSLIB.Quaternion.prototype.normalize = function() {
 36   var l = Math.sqrt(this.x * this.x + this.y * this.y + this.z * this.z + this.w * this.w);
 37   if (l === 0) {
 38     this.x = 0;
 39     this.y = 0;
 40     this.z = 0;
 41     this.w = 1;
 42   } else {
 43     l = 1 / l;
 44     this.x = this.x * l;
 45     this.y = this.y * l;
 46     this.z = this.z * l;
 47     this.w = this.w * l;
 48   }
 49 };
 50 
 51 /**
 52  * Convert this quaternion into its inverse.
 53  */
 54 ROSLIB.Quaternion.prototype.inverse = function() {
 55   this.conjugate();
 56   this.normalize();
 57 };
 58 
 59 /**
 60  * Set the values of this quaternion to the product of itself and the given quaternion.
 61  *
 62  * @param q the quaternion to multiply with
 63  */
 64 ROSLIB.Quaternion.prototype.multiply = function(q) {
 65   this.x = this.x * q.w + this.y * q.z - this.z * q.y + this.w * q.x;
 66   this.y = -this.x * q.z + this.y * q.w + this.z * q.x + this.w * q.y;
 67   this.z = this.x * q.y - this.y * q.x + this.z * q.w + this.w * q.z;
 68   this.w = -this.x * q.x - this.y * q.y - this.z * q.z + this.w * q.w;
 69 };
 70 
 71 /**
 72  * Clone a copy of this quaternion.
 73  *
 74  * @returns the cloned quaternion
 75  */
 76 ROSLIB.Quaternion.prototype.clone = function() {
 77   return new ROSLIB.Quaternion(this);
 78 };
