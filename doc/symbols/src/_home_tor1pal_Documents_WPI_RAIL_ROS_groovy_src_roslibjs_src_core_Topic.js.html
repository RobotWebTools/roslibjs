  5 /**
  6  * Publish and/or subscribe to a topic in ROS.
  7  *
  8  * Emits the following events:
  9  *  * 'warning' - if there are any warning during the Topic creation
 10  *  * 'message' - the message data from rosbridge
 11  *
 12  * @constructor
 13  * @param options - object with following keys:
 14  *   * ros - the ROSLIB.Ros connection handle
 15  *   * name - the topic name, like /cmd_vel
 16  *   * messageType - the message type, like 'std_msgs/String'
 17  *   * compression - the type of compression to use, like 'png'
 18  *   * throttle_rate - the rate at which to throttle the topics
 19  */
 20 ROSLIB.Topic = function(options) {
 21   options = options || {};
 22   this.ros = options.ros;
 23   this.name = options.name;
 24   this.messageType = options.messageType;
 25   this.isAdvertised = false;
 26   this.compression = options.compression || 'none';
 27   this.throttle_rate = options.throttle_rate || 0;
 28 
 29   // Check for valid compression types
 30   if (this.compression && this.compression !== 'png' && this.compression !== 'none') {
 31     this.emit('warning', this.compression
 32         + ' compression is not supported. No comression will be used.');
 33   }
 34 
 35   // Check if throttle rate is negative
 36   if (this.throttle_rate < 0) {
 37     this.emit('warning', this.throttle_rate + ' is not allowed. Set to 0');
 38     this.throttle_rate = 0;
 39   }
 40 };
 41 ROSLIB.Topic.prototype.__proto__ = EventEmitter2.prototype;
 42 
 43 /**
 44  * Every time a message is published for the given topic, the callback
 45  * will be called with the message object.
 46  *
 47  * @param callback - function with the following params:
 48  *   * message - the published message
 49  */
 50 ROSLIB.Topic.prototype.subscribe = function(callback) {
 51   var that = this;
 52 
 53   this.on('message', function(message) {
 54     callback(message);
 55   });
 56 
 57   this.ros.on(this.name, function(data) {
 58     var message = new ROSLIB.Message(data);
 59     that.emit('message', message);
 60   });
 61 
 62   this.ros.idCounter++;
 63   var subscribeId = 'subscribe:' + this.name + ':' + this.ros.idCounter;
 64   var call = {
 65     op : 'subscribe',
 66     id : subscribeId,
 67     type : this.messageType,
 68     topic : this.name,
 69     compression : this.compression,
 70     throttle_rate : this.throttle_rate
 71   };
 72 
 73   this.ros.callOnConnection(call);
 74 };
 75 
 76 /**
 77  * Unregisters as a subscriber for the topic. Unsubscribing will remove
 78  * all subscribe callbacks.
 79  */
 80 ROSLIB.Topic.prototype.unsubscribe = function() {
 81   this.ros.removeAllListeners([ this.name ]);
 82   this.ros.idCounter++;
 83   var unsubscribeId = 'unsubscribe:' + this.name + ':' + this.ros.idCounter;
 84   var call = {
 85     op : 'unsubscribe',
 86     id : unsubscribeId,
 87     topic : this.name
 88   };
 89   this.ros.callOnConnection(call);
 90 };
 91 
 92 /**
 93  * Registers as a publisher for the topic.
 94  */
 95 ROSLIB.Topic.prototype.advertise = function() {
 96   this.ros.idCounter++;
 97   var advertiseId = 'advertise:' + this.name + ':' + this.ros.idCounter;
 98   var call = {
 99     op : 'advertise',
100     id : advertiseId,
101     type : this.messageType,
102     topic : this.name
103   };
104   this.ros.callOnConnection(call);
105   this.isAdvertised = true;
106 };
107 
108 /**
109  * Unregisters as a publisher for the topic.
110  */
111 ROSLIB.Topic.prototype.unadvertise = function() {
112   this.ros.idCounter++;
113   var unadvertiseId = 'unadvertise:' + this.name + ':' + this.ros.idCounter;
114   var call = {
115     op : 'unadvertise',
116     id : unadvertiseId,
117     topic : this.name
118   };
119   this.ros.callOnConnection(call);
120   this.isAdvertised = false;
121 };
122 
123 /**
124  * Publish the message.
125  *
126  * @param message - A ROSLIB.Message object.
127  */
128 ROSLIB.Topic.prototype.publish = function(message) {
129   if (!this.isAdvertised) {
130     this.advertise();
131   }
132 
133   this.ros.idCounter++;
134   var publishId = 'publish:' + this.name + ':' + this.ros.idCounter;
135   var call = {
136     op : 'publish',
137     id : publishId,
138     topic : this.name,
139     msg : message
140   };
141   this.ros.callOnConnection(call);
142 };
