  6 /**
  7  * A Visual element in a URDF.
  8  * 
  9  * @constructor
 10  * @param options - object with following keys:
 11  *  * xml - the XML element to parse
 12  */
 13 ROSLIB.UrdfVisual = function(options) {
 14   var that = this;
 15   var options = options || {};
 16   var xml = options.xml;
 17   this.origin = null;
 18   this.geometry = null;
 19   this.material = null;
 20 
 21   /**
 22    * Initialize the element with the given XML node.
 23    * 
 24    * @param xml - the XML element to parse
 25    */
 26   var initXml = function(xml) {
 27     // origin
 28     var origins = xml.getElementsByTagName('origin');
 29     if (origins.length === 0) {
 30       // use the identity as the default
 31       that.origin = new ROSLIB.Pose();
 32     } else {
 33       // check the XYZ
 34       var xyz = xml.getAttribute('xyz');
 35       if (!xyz) {
 36         // use the default values
 37         var position = new ROSLIB.Vector3();
 38       } else {
 39         var xyz = xyz.split(' ');
 40         var position = new ROSLIB.Vector3({
 41           x : parseFloat(xyz[0]),
 42           y : parseFloat(xyz[1]),
 43           z : parseFloat(xyz[2])
 44         });
 45       }
 46 
 47       // check the RPY
 48       rpy = xml.getAttribute('rpy');
 49       if (!rpy) {
 50         // use the default values
 51         var orientation = new ROSLIB.Quaternion();
 52       } else {
 53         var rpy = rpy.split(' ');
 54         // convert from RPY
 55         var roll = parseFloat(rpy[0]);
 56         var pitch = parseFloat(rpy[1]);
 57         var yaw = parseFloat(rpy[2]);
 58         var phi = roll / 2.0;
 59         var the = pitch / 2.0;
 60         var psi = yaw / 2.0;
 61         var x = Math.sin(phi) * Math.cos(the) * Math.cos(psi) - Math.cos(phi) * Math.sin(the)
 62             * Math.sin(psi);
 63         var y = Math.cos(phi) * Math.sin(the) * Math.cos(psi) + Math.sin(phi) * Math.cos(the)
 64             * Math.sin(psi);
 65         var z = Math.cos(phi) * Math.cos(the) * Math.sin(psi) - Math.sin(phi) * Math.sin(the)
 66             * Math.cos(psi);
 67         var w = Math.cos(phi) * Math.cos(the) * Math.cos(psi) + Math.sin(phi) * Math.sin(the)
 68             * Math.sin(psi);
 69 
 70         var orientation = new ROSLIB.Quaternion({
 71           x : x,
 72           y : y,
 73           z : z,
 74           w : w
 75         });
 76         orientation.normalize();
 77       }
 78       that.origin = new ROSLIB.Pose({
 79         position : position,
 80         orientation : orientation
 81       });
 82     }
 83 
 84     // geometry
 85     var geoms = xml.getElementsByTagName('geometry');
 86     if (geoms.length > 0) {
 87       var shape = null;
 88       // check for the shape
 89       for (n in geoms[0].childNodes) {
 90         var node = geoms[0].childNodes[n];
 91         if (node.nodeType === 1) {
 92           shape = node;
 93           break;
 94         }
 95       }
 96       // check the type
 97       var type = shape.nodeName;
 98       if (type === 'sphere') {
 99         that.geometry = new ROSLIB.UrdfSphere({
100           xml : shape
101         });
102       } else if (type === 'box') {
103         that.geometry = new ROSLIB.UrdfBox({
104           xml : shape
105         });
106       } else if (type === 'cylinder') {
107         that.geometry = new ROSLIB.UrdfCylinder({
108           xml : shape
109         });
110       } else if (type === 'mesh') {
111         that.geometry = new ROSLIB.UrdfMesh({
112           xml : shape
113         });
114       } else {
115         console.warn('Unknown geometry type ' + type);
116       }
117     }
118 
119     // material
120     var materials = xml.getElementsByTagName('material');
121     if (materials.length > 0) {
122       that.material = new ROSLIB.UrdfMaterial({
123         xml : materials[0]
124       });
125     }
126   };
127 
128   // pass it to the XML parser
129   initXml(xml);
130 };
