  5 /**
  6  * A ServiceRequest is passed into the service call.
  7  *
  8  * @constructor
  9  * @param options - possible keys include:
 10  *   * values - object matching the values of the request part from the .srv file.
 11  */
 12 ROSLIB.ServiceRequest = function(options) {
 13   var that = this;
 14   var options = options || {};
 15   var values = options.values;
 16 
 17   if (values) {
 18     Object.keys(values).forEach(function(name) {
 19       that[name] = values[name];
 20     });
 21   }
 22 };
