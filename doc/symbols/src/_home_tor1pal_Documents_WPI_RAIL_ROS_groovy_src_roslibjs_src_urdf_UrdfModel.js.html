  6 /**
  7  * A URDF Model can be used to parse a given URDF into the appropriate elements. 
  8  * 
  9  * @constructor
 10  * @param options - object with following keys:
 11  *  * xml - the XML element to parse
 12  *  * string - the XML element to parse as a string
 13  */
 14 ROSLIB.UrdfModel = function(options) {
 15   var that = this;
 16   var options = options || {};
 17   var xml = options.xml;
 18   var string = options.string;
 19 
 20   this.name;
 21   this.materials = [];
 22   this.links = [];
 23 
 24   /**
 25    * Initialize the model with the given XML node.
 26    * 
 27    * @param xml - the XML element to parse
 28    */
 29   var initXml = function(xml) {
 30     // get the robot tag
 31     var robotXml = xml.evaluate('//robot', xml, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
 32 
 33     // get the robot name
 34     that.name = robotXml.getAttribute('name');
 35 
 36     // parse all the visual elements we need
 37     for (n in robotXml.childNodes) {
 38       var node = robotXml.childNodes[n];
 39       if (node.tagName === 'material') {
 40         var material = new ROSLIB.UrdfMaterial({
 41           xml : node
 42         });
 43         // make sure this is unique
 44         if (that.materials[material.name]) {
 45           console.warn('Material ' + material.name + 'is not unique.');
 46         } else {
 47           that.materials[material.name] = material;
 48         }
 49       } else if (node.tagName === 'link') {
 50         var link = new ROSLIB.UrdfLink({
 51           xml : node
 52         });
 53         // make sure this is unique
 54         if (that.links[link.name]) {
 55           console.warn('Link ' + link.name + ' is not unique.');
 56         } else {
 57           // check for a material
 58           if (link.visual && link.visual.material) {
 59             if (that.materials[link.visual.material.name]) {
 60               link.visual.material = that.materials[link.visual.material.name];
 61             } else if (link.visual.material) {
 62               that.materials[link.visual.material.name] = link.visual.material;
 63             }
 64           }
 65 
 66           // add the link
 67           that.links[link.name] = link;
 68         }
 69       }
 70     }
 71   };
 72 
 73   // check if we are using a string or an XML element
 74   if (string) {
 75     // parse the string
 76     var parser = new DOMParser();
 77     xml = parser.parseFromString(string, 'text/xml');
 78   }
 79   // pass it to the XML parser
 80   initXml(xml);
 81 };
