  6 /**
  7  * A Box element in a URDF.
  8  * 
  9  * @constructor
 10  * @param options - object with following keys:
 11  *  * xml - the XML element to parse
 12  */
 13 ROSLIB.UrdfBox = function(options) {
 14   var that = this;
 15   var options = options || {};
 16   var xml = options.xml;
 17   this.dimension = null;
 18   this.type = null;
 19 
 20   /**
 21    * Initialize the element with the given XML node.
 22    * 
 23    * @param xml - the XML element to parse
 24    */
 25   var initXml = function(xml) {
 26     this.type = ROSLIB.URDF_BOX;
 27 
 28     // parse the string
 29     var xyz = xml.getAttribute('size').split(' ');
 30     that.dimension = new ROSLIB.Vector3({
 31       x : parseFloat(xyz[0]),
 32       y : parseFloat(xyz[1]),
 33       z : parseFloat(xyz[2])
 34     });
 35   };
 36 
 37   // pass it to the XML parser
 38   initXml(xml);
 39 };
